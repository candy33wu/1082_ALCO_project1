string toBin(string str,int n)
{
	bool neg = false;
	string ans = "00000000000000000000000000000000";
	string s;
	if(str[0]=='x')
		str=str.substr(1, str.length());
	if (str[0] == '-'){
		ans = "11111111111111111111111111111111";
		str = str.substr(1, str.length());
		neg = true;
	}

	int buffer;
	stringstream ss;
	ss << str;
	ss >> buffer;
	ss.str("");
	ss.clear();
	long long int total = 0, m = 1;
	
	while (buffer != 0) {
		total += ((buffer % 2) * m);
		buffer /= 2;
		m *= 10; 
	}
	ss << total;
	ss >> s;
	if (neg) {
		for (int i = 0; i < s.length(); ++i) {
			if (s[i] == '1')
				s[i] = '0';
			else
				s[i] = '1';
		}
		for (int i = s.length()-1; i >0 ; --i) {
			if (s[i] == '1')
				s[i] = '0';
			else {
				s[i] = '1';
				break;
			}
		}
	}
	ans += s;
	return ans.substr(ans.length() - n, ans.length());
}

string findoffset(string str,int pc,int limit) {
	int offset = 0;
	string ans;
	for (int i = 0; i < label.size(); ++i) {
		if (str == label[i]) {
			offset = 4*(lbnum[i] - pc);
			break;
		}
	}
	ans = to_string(offset);
	ans = toBin(ans, limit);
	return ans;
}
